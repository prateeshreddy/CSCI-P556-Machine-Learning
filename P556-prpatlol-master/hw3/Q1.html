def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def sigmoid_derivative(x):
    return sigmoid(x) * (1 - sigmoid(x))

def ae(y, yhat):
    return np.abs(y - yhat)

def bce(y, yhat):
    return np.log(yhat + 1e-20) * y + np.log(1-yhat + 1e-20) * (1-y)

def bcel(y, yhat):
    return -np.mean(np.log(yhat + 1e-20) * y + np.log(1-yhat + 1e-20) * (1-y))

class Perceptron:
    def __init__(self, n_predictors, n_responses, n_hidden, init="normal", init_seed=123):
        n_predictors += 1
        self.n_predictors, self.n_responses, self.n_hidden = n_predictors, n_responses, n_hidden
        # Init parameters and scale them to [-1, 1]
        if init_seed > -1:
            np.random.seed(init_seed)
        if init == "normal":
            self.l1_w = np.random.normal(size=[n_hidden, n_predictors])
            self.l2_w = np.random.normal(size=[n_responses, n_hidden])
            # Scale to [-1, 1]
            self.l1_w /= np.max(np.abs(self.l1_w))
            self.l2_w /= np.max(np.abs(self.l2_w))
        elif init == "uniform":
            self.l1_w = np.random.uniform(-1, 1, size=[n_hidden, n_predictors])
            self.l2_w = np.random.uniform(-1, 1, size=[n_responses, n_hidden])
        else:
            raise NotImplementedError(init)
        # Init momentum
        self.l1_w_m = np.zeros([n_hidden, n_predictors])
        self.l2_w_m = np.zeros([n_responses, n_hidden])
        # Init activations and derivatives
        self.l1_act = sigmoid
        self.l2_act = sigmoid
        self.l1_act_drv = sigmoid_derivative
        self.l2_act_drv = sigmoid_derivative
        
    def forward(self, X):
        self.l0_z = np.concatenate([X, np.ones([X.shape[0], 1])], axis=1)
        self.l1_a = np.matmul(self.l0_z, np.transpose(self.l1_w))
        self.l1_z = self.l1_act(self.l1_a)
        self.l2_a = np.matmul(self.l1_z, np.transpose(self.l2_w))
        self.l2_z = self.l2_act(self.l2_a)
        return self.l2_z
    
    def backward(self, Y, Yhat):
        l2_dlt = Yhat - Y
        l2_w_grad = self.get_w_grad(self.l2_w, l2_dlt, self.l1_z)
        l1_dlt = self.get_dlt(self.l1_act_drv, self.l1_a, self.l2_w, l2_dlt)
        l1_w_grad = self.get_w_grad(self.l1_w, l1_dlt, self.l0_z)
        self.l2_w_grad = l2_w_grad
        self.l1_w_grad = l1_w_grad
        
    def get_dlt(self, act_drv, a, w, dlt):
        new_dlt = np.zeros(a.shape)
        for n in range(dlt.shape[0]):
            for j in range(w.shape[1]):
                summ = 0
                for k in range(w.shape[0]):
                    summ += w[k,j] * dlt[n,k]
                new_dlt[n,j] = act_drv(a)[n,j] * summ
        return new_dlt
    
    def get_w_grad(self, w, dlt, z):
        w_grad = np.zeros(w.shape)
        for n in range(dlt.shape[0]):
            for j in range(w.shape[0]):
                for i in range(w.shape[1]):
                    w_grad[j,i] += dlt[n,j] * z[n,i]
        return w_grad
    
    def step(self, lr, alpha=0.0):
        self.l2_w_m = alpha * self.l2_w_m - self.l2_w_grad
        self.l1_w_m = alpha * self.l1_w_m - self.l1_w_grad
        self.l2_w += lr * self.l2_w_m
        self.l1_w += lr * self.l1_w_m
        
    def optimize(self, X, Y, loss, n_epochs=100, threshold=None, mbatch_size=1, lr=0.001, alpha=0.0, stochastic=True):
        losses = []
        n_samples = X.shape[0]
        np.random.seed(123)
        done = False
        epoch = 0
        while not done:
            epoch += 1
            perm = np.random.choice(n_samples, size=n_samples, replace=False)
            if not stochastic:
                perm = sorted(perm)
            n_mbatches = n_samples // mbatch_size
            for j in range(n_mbatches):
                start = mbatch_size * j
                end = mbatch_size * (j + 1)
                Yhat = self.forward(X[perm[start:end]])
                self.backward(Y[perm[start:end]], Yhat)
                self.step(lr, alpha)
            Yhat = self.forward(X)
            Yhat_ae = ae(Y, Yhat)
            losses += [loss(Y, Yhat)]
            done = False
            if not threshold is None:
                # Done when number of predictions with absolute error > threshold is 0
                done = len(Yhat_ae[Yhat_ae > threshold]) == 0
            if n_epochs > -1:
                done = done or epoch >= n_epochs
            if epoch % 1000 == 0:
                print("Epoch %d Loss = %.5f" % (epoch, losses[epoch-1]))
        self.losses = losses
        return epoch
